                258         function [labels,scores,cost] = predict(this,X,varargin)
                259         %PREDICT Predict response of the model.
                260         %   [LABEL,POSTERIOR,COST]=PREDICT(OBJ,X) returns predicted class labels
                261         %   LABEL, posterior probabilities POSTERIOR and misclassification costs
                262         %   COST for model OBJ and a matrix of predictors X. X must be a numeric
                263         %   matrix of size N-by-P, where P is the number of predictors used for
                264         %   training this model. Classification labels LABEL have the same type as
                265         %   Y used for training. Posterior probabilities POSTERIOR are an N-by-K
                266         %   numeric matrix for N observations and K classes. COST is an N-by-K
                267         %   matrix with predicted misclassification costs per class. The predicted
                268         %   label is assigned to the class with the minimal misclassification cost.
                269         %
                270         %   See also classreg.learning.classif.ClassificationModel.
                271   
                272             % Empty data
                273             if isempty(X)
                274                 [labels,scores,cost] = predictEmptyX(this,X);
                275                 return;
                276             end
                277             
                278             % Get scores from the compact class
                279             scores = score(this,X,varargin{:});
                280             
                281             % Transform scores and find the most probable class
                282             [labels,scores,cost] = this.LabelPredictor(this.ClassNames,...
                283                 this.Prior,this.Cost,scores,this.PrivScoreTransform);
                284         end
